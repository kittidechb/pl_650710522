OOP Concept แนวคิดของการเขียนโปรแกรมเชิงวัตถุ

I. เขียนโปรแกรมในภาษา Java, C++, Python เปรียบเทียบผลรัน

Java

class Animal {
    public Animal() {
        System.out.println("Create animal");       
    }
    public void sound() {
        System.out.println("Animal makes a sound");       
    }
    @Override
    protected void finalize() throws Throwable {
        System.out.println("Delete animal");
    }
}
class Dog extends Animal {
    public Dog() {
        System.out.println("Dog");
    }
    @Override
    public void sound() {
        System.out.println("Dog barks");
    }
    @Override
    protected void finalize() throws Throwable {
        System.out.println("Delete dog");
    }
}
class Cat extends Animal {
    public Cat() {
        System.out.println("Cat");
    }
    @Override
    public void sound() {
        System.out.println("Cat meows");
    }
    @Override
    protected void finalize() throws Throwable {
        System.out.println("Delete cat");
    }
}
public class Main {
    public static void main(String[] args) {
        Animal a = new Animal();
            a.sound();
            Dog d = new Dog();
            d.sound();
            Cat c = new Cat();
            c.sound();
            a = null;
            d = null;
            c = null;
    }
}

II. ทำความเข้าใจส่วนประกอบของโค้ดเปรียบเทียบกับ concept ของ OOP ต่อไปนี้ โดยยกตัวอย่างส่วนประกอบของโค้ดประกอบการอธิบาย

1. Class (คลาส)

คำอธิบาย: คลาสเป็นแบบแผนหรือพิมพ์เขียวที่กำหนดคุณสมบัติ (Attributes) และพฤติกรรม (Methods) ของวัตถุ (Objects) ที่สร้างขึ้นจากคลาสนั้น

Java

class Animal {
    void sound() { //Method
        System.out.println("Animal makes a sound");
        }
}

C++

class Animal {
    public:
        void sound() { //Method
            std::cout << "Animal makes a sound" << std::endl;
        }
};

Python

class Animal:
    def sound(self): #Method
        print("Animal makes a sound")

2. Object, Instance (ออบเจ็กต์, อินสแตนซ์)

คำอธิบาย: ออบเจ็กต์คือการสร้างอิงจากคลาส ซึ่งมีคุณสมบัติและพฤติกรรมที่สามารถทำงานได้ โดยออบเจ็กต์แต่ละตัวสามารถมีค่าที่แตกต่างกันในคุณสมบัติของมัน

Java

Animal a = new Animal();  
// สร้างออบเจ็กต์ a จากคลาส Animal
a.sound(); 
// เรียกใช้เมธอด

C++

Animal* a = new Animal();                
// สร้างออบเจ็กต์ a จากคลาส Animal
a.sound();      
// เรียกใช้เมธอด

Python

a = Animal("Elephant")
# สร้างออบเจ็กต์ a จากคลาส Animal
a.sound()             
# เรียกใช้เมธอด

3. Subclass, Derived Class (คลาสย่อย, คลาสที่สืบทอด)

คำอธิบาย: คลาสย่อย (หรือคลาสลูก) เป็นคลาสที่สืบทอดคุณสมบัติและพฤติกรรมจากคลาสแม่ (Superclass) และสามารถปรับแต่งหรือขยายพฤติกรรมของตนเองได้

Java

class Dog extends Animal {  
//Dogเป็นคลาสย่อยที่สืบทอดจากAnimal
void sound() {    // เขียนทับเมธอด sound()
    System.out.println("Dog barks");
    }
}

C++

class Dog : public Animal { //DogสืบทอดจากAnimal
    public:
        void sound() override {   // เขียนทับเมธอด sound()
            std::cout << "Dog barks" << std::endl;
        }
    };

Python

class Dog(Animal):  # Dog สืบทอดจาก Animal
def sound(self):  # เขียนทับเมธอด sound()
    print("Dog barks")

4. Message (ข้อความ)

คำอธิบาย: ข้อความใน OOP หมายถึงการเรียกใช้เมธอดของออบเจ็กต์ ซึ่งอาจทำให้เกิดการเปลี่ยนแปลงในคุณสมบัติหรือการทำงานของออบเจ็กต์

Java

 // ส่งข้อความไปที่ออบเจ็กต์ a เพื่อเรียกใช้เมธอด sound()

C++

a.sound(); 
// ส่งข้อความไปที่ออบเจ็กต์ a เพื่อเรียกใช้เมธอด sound()

Python

a.sound() 
# ส่งข้อความไปที่ออบเจ็กต์ a เพื่อเรียกใช้เมธอด sound()

5. Inheritance (การสืบทอด)

คำอธิบาย: การสืบทอดเป็นแนวคิดที่ช่วยให้สามารถสร้างคลาสใหม่จากคลาสที่มีอยู่แล้ว โดยคลาสใหม่จะสืบทอดคุณสมบัติและพฤติกรรมจากคลาสเก่า ทำให้สามารถใช้โค้ดซ้ำได้

Java

class Cat extends Animal { //CatสืบทอดจากAnimal
    void sound() { // เขียนทับเมธอด sound()
        System.out.println("Cat meows");
    }
}

C++

class Cat : public Animal { //CatสืบทอดจากAnimal
    public:
        void sound() override { 
            //เขียนทับเมธอดsound()
            std::cout << "Cat meows" << std::endl;
        }
};

Python

class Cat(Animal):  # Cat สืบทอดจาก Animal
def sound(self):  # เขียนทับเมธอด sound()
    print("Cat meows")

6. Polymorphism (พอลิเมอร์ฟิซึม)

คำอธิบาย: พอลิเมอร์ฟิซึมคือความสามารถของ OOP ที่อนุญาตให้ใช้เมธอดเดียวกันในหลาย ๆ รูปแบบ โดยอาจขึ้นอยู่กับประเภทของออบเจ็กต์ที่เรียกใช้ ซึ่งช่วยให้โปรแกรมมีความยืดหยุ่นมากขึ้น

Java

Animal myDog = new Dog();  // สร้างออบเจ็กต์ Dog แต่เก็บไว้ในตัวแปรชนิด Animal
Animal myCat = new Cat();  // สร้างออบเจ็กต์ Cat แต่เก็บไว้ในตัวแปรชนิด Animal
myDog.sound();  // Dog barks
myCat.sound();  // Cat meows

C++

Animal* myDog = new Dog();  // สร้างออบเจ็กต์ Dog
Animal* myCat = new Cat();  // สร้างออบเจ็กต์ Cat
myDog->sound();  // Dog barks
myCat->sound();  // Cat meows
delete myDog;    // คืนหน่วยความจำ
delete myCat;    // คืนหน่วยความจำ

Python

my_dog = Dog()  # สร้างออบเจ็กต์ Dog
my_cat = Cat()  # สร้างออบเจ็กต์ Cat
my_dog.sound()  # Dog barks
my_cat.sound()   # Cat meows

III. abstract class คืออะไร ในตัวอย่างข้อ I ที่ให้ class ใดคือ abstract class และมีหลักการสร้างอย่างไร

คำอธิบาย: คลาสนามธรรม (Abstract Class) คือคลาสที่ไม่สามารถสร้างออบเจ็กต์ได้โดยตรง และมักจะถูกใช้เป็นแม่แบบสำหรับคลาสย่อย (Subclass) เพื่อกำหนดรูปแบบ (Interface) หรือพฤติกรรมที่จำเป็นต้องมีในคลาสที่สืบทอด โดยคลาสนามธรรมสามารถมีเมธอดที่เป็นนามธรรม (Abstract Method) ซึ่งเป็นเมธอดที่ไม่มีการนิยาม (Implementation) ในคลาสนามธรรม แต่ต้องถูกเขียนทับ (Override) ในคลาสย่อยที่สืบทอดจากคลาสนามธรรมนี้

การสร้าง Abstract Class:

  • ใน Java ใช้คำว่า abstract ในการประกาศคลาสและเมธอด
  • ใน C++ ใช้ virtual และ = 0 เพื่อกำหนดเมธอดที่เป็นนามธรรม
  • ใน Python ใช้โมดูล abc เพื่อกำหนดคลาสนามธรรม

ตัวอย่างจากข้อ I ในตัวอย่างที่ให้มา คลาส Animal สามารถถูกทำให้เป็นคลาสนามธรรมได้ โดยการเปลี่ยนแปลงการประกาศเป็นคลาสนามธรรม และสร้างเมธอด sound() เป็นนามธรรม

Java

abstract class Animal { // กำหนดให้ Animal เป็น abstract class
    abstract void sound(); // เมธอดที่ไม่มีการนิยาม
}
class Dog extends Animal {
     sound() { // เขียนทับเมธอด sound()
        System.out.println("Dog barks");
    }
}
class Cat extends Animal {
    void sound() { // เขียนทับเมธอด sound()
        System.out.println("Cat meows");
    }
}

C++

class Animal { // คลาส Animal เป็น base class
    public:
        virtual void sound() = 0; // เมธอดที่เป็นนามธรรม
};
class Dog : public Animal {
    public:
        void sound() override { // เขียนทับเมธอด sound()
            std::cout << "Dog barks" << std::endl;
        }
};
    class Cat : public Animal {
        public:
            void sound() override { // เขียนทับเมธอด sound()
                std::cout << "Cat meows" << std::endl;
        }
};

Python

my_dog = Dog()  # สร้างออบเจ็กต์ Dog
my_cat = Cat()  # สร้างออบเจ็กต์ Cat
my_dog.sound()  # Dog barks
my_cat.sound()
# Cat meowsfrom abc import ABC, abstractmethod

class Animal(ABC): 
# กำหนดให้ Animal เป็น abstract class
    @abstractmethod
    def sound(self): # เมธอดที่เป็นนามธรรม
        pass

class Dog(Animal):
    def sound(self): # เขียนทับเมธอด sound()
        print("Dog barks")

class Cat(Animal):
    def sound(self): # เขียนทับเมธอด sound()
        print("Cat meows")
